GROUP BY

players %>%
  mutate(year=as.numeric(substring(birthday, 1, 4))) %>%
  group_by(year) %>%
  summarise(weight=mean(height)) %>%
  ggplot(aes(year, weight)) +
  geom_line(colour="darkorange") +
  theme_minimal()

players %>%
  mutate(year=as.numeric(substring(birthday, 1, 4))) %>%
  group_by(year) %>%
  summarise(nsample=n()) %>%
  ggplot(aes(year, nsample)) +
  geom_line(colour="darkorange") +
  theme_minimal()

players %>%
  mutate(year=as.numeric(substring(birthday, 1, 4))) %>%
  group_by(year) %>%
  summarise(weight=mean(height), nsamples=n()) %>%
  filter(nsamples >= 200) %>%
  ggplot(aes(year, weight)) +
  geom_line(colour="darkorange") +
  theme_minimal()
